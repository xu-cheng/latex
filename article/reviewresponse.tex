\RequirePackage{mdframed}

\newcounter{@reviewer}
\newcounter{@comment}[@reviewer]

\newcommand{\newreviewer}{%
    \stepcounter{@reviewer}
    \phantomsection
    \bookmark[dest=\@currentHref,rellevel=1,keeplevel]{Reviewer \arabic{@reviewer}}
    {
        \large
        \vskip\topskip
        \noindent\bfseries\underline{Reviewer \arabic{@reviewer}}
        \par
        \vskip\topskip
    }
}
\newenvironment{comment}{%
    \stepcounter{@comment}
    \phantomsection
    \bookmark[dest=\@currentHref,rellevel=2,keeplevel]{Comment \arabic{@comment}}
    \noindent
    R\arabic{@reviewer}.\arabic{@comment}:
    \itshape
}{%
    \par
}
\surroundwithmdframed[
    backgroundcolor=black!8,
    hidealllines=true,
    innertopmargin=\topskip,
    innerbottommargin=\topskip,
    skipabove=0,
    skipbelow=0
]{comment}
\newenvironment{response}{%
    \phantomsection
    \bookmark[dest=\@currentHref,rellevel=3,keeplevel]{Response}
    \noindent\textbf{Response:}
}{%
    \par
    \vskip\topskip
}
